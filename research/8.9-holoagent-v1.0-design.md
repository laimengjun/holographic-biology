- Role adoption 完美

### 1.2 v0.2.6 的局限

**局限 1：单文件部署限制**
- v0.2.6 是单 Python 文件 ~500 行
- 不支持复杂的多文件架构
- 不支持插件化

**局限 2：只有 handoff() 一种协作模式**
- 不能并行
- 不能投票
- 不能共识

**局限 3：无记忆系统**
- 每次启动 Agent 是"全新"的
- 没有跨 session 经验保留
- 不能学习

**局限 4：依赖 OpenClaw**
- 必须使用 `openclaw agents` 命令
- 不能独立部署
- 不能跨 OpenClaw 实例

**局限 5：可观测性不足**
- 简单的 trace/log
- 无 metrics（QPS / 错误率 / 延迟分布）
- 无 dashboard

**局限 6：无安全机制**
- 无身份认证
- 无消息签名
- 无访问控制

### 4.4 工作流引擎

```python
class Workflow:
    """DAG 工作流引擎"""
    
    def __init__(self, name: str):
        self.name = name
        self.stages = []
    
    def add_stage(self, name: str, agents: List[str], 
                   mode: str = "parallel") -> 'Workflow':
        """添加工作流阶段"""
        self.stages.append({
            "name": name,
            "agents": agents,
            "mode": mode  # sequential | parallel | vote | consensus
        })
        return self
    
    async def execute(self, initial_payload: dict) -> dict:
        """执行工作流"""
        payload = initial_payload
        for stage in self.stages:
            if stage['mode'] == 'sequential':
                for agent in stage['agents']:
                    payload = await collaborate.request(agent, stage['name'], payload)
            elif stage['mode'] == 'parallel':
                responses = await collaborate.gather(stage['agents'], stage['name'], payload)
                payload = self.merge_responses(responses)
            elif stage['mode'] == 'vote':
                result = await collaborate.vote(stage['agents'], stage['name'], payload)
                payload = result.winning
            elif stage['mode'] == 'consensus':
                result = await collaborate.consensus(stage['agents'], stage['name'], payload)
                payload = result.agreed
        return payload
```

### 5.2 记忆 API

```python
class Memory:
    """3 层记忆管理"""
    
    def __init__(self, agent_id: str, config: MemoryConfig):
        self.agent_id = agent_id
        self.working = WorkingMemory(agent_id, config.redis_url)
        self.short = ShortTermMemory(agent_id, config.sqlite_path)
        self.long = LongTermMemory(agent_id, config.postgres_url, config.pgvector_url)
    
    # 工作记忆
    async def set(self, key: str, value: Any, ttl: int = None):
        await self.working.set(key, value, ttl)
    
    async def get(self, key: str) -> Any:
        return await self.working.get(key)
    
    # 短期记忆
    async def append(self, role: str, content: str, 
                      metadata: dict = None):
        await self.short.append(role, content, metadata)
    
    async def search(self, query: str, top_k: int = 10) -> List[MemoryItem]:
        return await self.short.search(query, top_k)
    
    # 长期记忆
    async def store(self, content: str, metadata: dict = None,
                     importance: float = 0.5):
        embedding = await self.embed(content)
        await self.long.store(embedding, content, metadata, importance)
    
    async def recall(self, query: str, top_k: int = 5) -> List[MemoryItem]:
        return await self.long.recall(query, top_k)
```

### 5.3 自动记忆流程

```
Agent 交互
  ↓
reflect() 评估每条消息
  ↓
importance 评分
  ├─ > 0.7 → store to long_term
  ├─ 0.3-0.7 → keep in short_term
  └─ < 0.3 → 丢弃
```

### 5.4 记忆检索算法

```python
async def recall(self, query: str, top_k: int = 5):
    # 1. Embedding 查询
    query_embedding = await self.embed(query)
    
    # 2. 向量相似度搜索
    results = await self.long.vector_search(
        embedding=query_embedding,
        top_k=top_k * 2  # 多检索一些
    )
    
    # 3. 重排序（考虑 importance + recency）
    reranked = sorted(
        results,
        key=lambda r: r.similarity * 0.6 + r.importance * 0.3 + r.recency * 0.1,
        reverse=True
    )[:top_k]
    
    return reranked
```

### 8.2 消息签名

```python
import hmac
import hashlib

def sign_message(message: dict, secret_key: str) -> str:
    """签名"""
    msg_str = json.dumps(message, sort_keys=True, separators=(',', ':'))
    return hmac.new(
        secret_key.encode(),
        msg_str.encode(),
        hashlib.sha256
    ).hexdigest()

def verify_signature(message: dict, signature: str, secret_key: str) -> bool:
    """验证签名"""
    expected = sign_message(message, secret_key)
    return hmac.compare_digest(expected, signature)
```

# 验证
def check_permission(agent_id, action, resource):
    acl = acls.get(agent_id)
    if not acl:
        return False
    if action in acl.get("read", []):
        return True
    if action in acl.get("write", []):
        return True
    return False
